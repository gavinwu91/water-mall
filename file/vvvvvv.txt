CREATE TABLE `sys_operate_log` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT COMMENT '日志ID',
  `system_code` varchar(128) NOT NULL COMMENT '系统编码',
  `system_name` varchar(256) NOT NULL COMMENT '系统名称',
  `operator_id` varchar(128) DEFAULT NULL COMMENT '操作人ID',
  `operator_account` varchar(128) DEFAULT NULL COMMENT '操作人账号',
  `operator_name` varchar(128) NOT NULL COMMENT '操作人姓名',
  `operator_mobile` varchar(64) DEFAULT NULL COMMENT '操作人联系方式',
  `operator_role_name` varchar(64) DEFAULT NULL COMMENT '操作人角色名称',
  `operator_ip` varchar(64) NOT NULL COMMENT '操作人IP',
  `operate_model` varchar(256) DEFAULT NULL COMMENT '操作模块',
  `operate_start_time` datetime NOT NULL COMMENT '操作开始时间（yyyy-MM-dd HH:mm:ss）',
  `operate_end_time` datetime NOT NULL COMMENT '操作结束时间（yyyy-MM-dd HH:mm:ss）',
  `operate_date` date NOT NULL COMMENT '操作日期（yyyy-MM-dd）',
  `operate_type` varchar(32) NOT NULL COMMENT '操作类别（add、modify、search）',
  `operate_expend_time` bigint(20) DEFAULT NULL COMMENT '操作耗时（毫秒）',
  `operate_url` varchar(256) DEFAULT NULL COMMENT '操作路径url',
  `operate_input` varchar(4000) NOT NULL COMMENT '操作入参',
  `operate_output` varchar(4000) DEFAULT NULL COMMENT '操作出参',
  `method` varchar(256) DEFAULT NULL COMMENT 'Controller方法',
  `request_method` varchar(64) DEFAULT NULL COMMENT '请求方法(GET/POST/PUT...)',
  `operate_result` varchar(32) DEFAULT NULL COMMENT '操作结果',
  `operate_result_desc` text COMMENT '操作结果描述(失败原因等)',
  PRIMARY KEY (`id`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=157275 DEFAULT CHARSET=utf8mb4 COMMENT='Sys operate log';
