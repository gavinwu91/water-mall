/**
 * 最优邮件翻译入口
 */
public static String newTrans(String text, String toLanguage, String llmTransUrl) {
    if (StrUtil.isBlank(text)) {
        return "";
    }

    String normalized = normalizeEmail(text);

    if (normalized.length() <= 2200) {
        System.out.println("文本较短，直接翻译，长度: " + normalized.length());
        return newTransApi(normalized, toLanguage, llmTransUrl);
    }

    List<String> chunks = smartSplit(normalized, 2000, 350);

    StringBuffer sb = new StringBuffer();
    for (int i = 0; i < chunks.size(); i++) {
        String chunk = chunks.get(i);
        System.out.println("翻译第 " + (i + 1) + "/" + chunks.size() + " 段，长度: " + chunk.length());

        long s = System.currentTimeMillis();
        String translated = newTransApi(chunk, toLanguage, llmTransUrl);
        sb.append(translated).append("\n\n");
        System.out.println("第 " + (i + 1) + " 段耗时: " + (System.currentTimeMillis() - s) / 1000 + " 秒");
    }
    return sb.toString().trim();
}

/**
 * 规范化清理 - 平衡版（推荐）
 * 保留邮件头信息，只清理真正重复无用的内容
 */
private static String normalizeEmail(String text) {
    if (StrUtil.isBlank(text)) return "";

    return text
            .replaceAll("[ \\t]+", " ")                    // 多个空格压缩成一个
            .replaceAll("\n{4,}", "\n\n\n")                // 最多保留2个空行
            .replaceAll("(?i)Disclaimer!.*?(?=\\n\\n|$)", "")   // 删除重复 Disclaimer
            .replaceAll("\\[NEW_01\\]", "")                // 删除 [NEW_01]
            .replaceAll("(?i)Bank Muscat:\\s*Official", "")   // 删除 Bank Muscat: Official
            .replaceAll("(?m)^\\s*_{3,}\\s*$", "")        // 删除分隔线 ________
            .lines()
            .map(String::trim)
            .filter(line -> !line.isBlank())
            .collect(Collectors.joining("\n"))
            .trim();
}

// 智能分段（保持不变）
private static List<String> smartSplit(String text, int maxChunkSize, int overlapSize) {
    List<String> chunks = new ArrayList<>();
    String[] paragraphs = text.split("\n\n");
    StringBuilder current = new StringBuilder();

    for (String para : paragraphs) {
        if (para.length() > maxChunkSize) {
            if (current.length() > 0) {
                chunks.add(current.toString());
                current = new StringBuilder();
            }
            for (int start = 0; start < para.length(); start += maxChunkSize - overlapSize) {
                int end = Math.min(start + maxChunkSize, para.length());
                chunks.add(para.substring(start, end));
            }
            continue;
        }

        if (current.length() + para.length() + 2 > maxChunkSize && current.length() > 0) {
            chunks.add(current.toString());
            int keep = Math.min(overlapSize, current.length());
            current = new StringBuilder(current.substring(current.length() - keep));
        }
        if (current.length() > 0) current.append("\n\n");
        current.append(para);
    }
    if (current.length() > 0) {
        chunks.add(current.toString());
    }
    return chunks;
}

private static String newTransApi(String text, String toLanguage, String llmTransUrl) {
    Map<String, Object> paramMap = new HashMap<>();
    paramMap.put("text", text);
    paramMap.put("source_lang", "auto");
    paramMap.put("target_lang", toLanguage);

    String post = HttpUtil.post(llmTransUrl, JSONUtil.toJsonStr(paramMap));
    LlmTransResult llmTransResult = JSONUtil.toBean(post, LlmTransResult.class);
    return llmTransResult.getTranslatedText();
}
